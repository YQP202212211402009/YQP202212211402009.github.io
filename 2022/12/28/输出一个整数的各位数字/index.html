Description

输入一个整数，从高位开始逐位输出它的各位数字。

Input

输入的整数可能大于1030

Output

给的数太大了，只能用字符串遍历了

#include<stdio.h>
#include<string.h>
main()
{
int r,i,b,j,count=0;
char a[1000];
scanf("%d",&r);
for(i=1;i<=r;i++)
{
count=0;
j=0;
    scanf("%s",&a);
    b=strlen(a);
    if(a[0]!='-')
    {
        while(a[j]!='\0')
        {
            if(count!=0)
            {
                printf(" ");
            }
            printf("%c",a[j]);
            count++;
            j++;
        }
        printf("\n");
    }
    else
    {
        j++;
        while(a[j]!='\0')
        {
            if(count!=0)
            {
                printf(" ");
            }
            printf("%c",a[j]);
            count++;
            j++;
        }
        printf("\n");
    }
}
}
