暑假作业

#来自录取通知书上的一个作业，题目的要求是写出现在日期与放假日期之间的时间 

#include<stdio.h>
main(){
int fangjia,i=0;
int sj1,sj2,sj3,rizi1; 
int fj1,fj2,fj3,rizi2;
int sum1=0,sum2=0,bs;
int yue[]={0,31,28,31,30,31,30,31,31,30,31,30,31};
printf("请输入当前日期：(格式xxxx-yy-zz) 如2008-1-2\n");
scanf("%d-%d-%d",&sj1,&sj2,&sj3);
    while(i<=sj2){
        sum1=sum1+yue[i];
        i++;
    }	
    rizi1=sj1*365+sum1+sj3;
    printf("请输入放假日期：(格式xxxx-yy-zz) 如2008-1-2\n");
     scanf("%d-%d-%d",&fj1,&fj2,&fj3);
     i=0;
     while(i<=fj2){
        sum2=sum2+yue[i];
        i++;
    }		
    rizi2=fj1*365+sum2+fj3;
    bs=rizi2-rizi1;

        printf("距离快乐假期还有%d天\n",rizi2-rizi1);
        else

    return 0;
}
