Description
输入 1 个四位数，将其加密后输出。方法是将该数每一位上的数字加9，然后除以10 取余，
做为该位上的新数字，最后将第1 位和第3 位上的数字互换，第2 位和第4 位上的数字互
换，组成加密后的新数。

Input
输入只有一个正整数

Output
输出加密后的新数

Sample Input
1257
Sample Output
The encrypted number is 4601 

#include <stdio.h>
int main()
{
    int number , temp,i =0 ,a[3];
    scanf("%d",&number);
        while(i < 4)
            {
                while(number > 0)
                {
                temp = number% 10 ;
                number = number /10 ;
                a[i] =temp ;
                i++ ;
                }

            }
            for(i = 0 ; i<4 ;i++)
            {
                a[i] += 9 ;
                a[i] = a[i] % 10 ;

            }
            temp = a[0] ;
            a[0] = a[2] ;
            a[2] = temp ;
            temp = a[1];
            a[1] = a[3] ;
            a[3] = temp ;
            number = a[0]+ a[1]*10 + a[2]*100 + a[3]*1000 ;
            printf("The encrypted number is %d",number);
            return 0;
}
