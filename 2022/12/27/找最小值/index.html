#题目
Description
输入一个正整数 repeat (0<repeat<10)，做repeat 次下列运算：
输入四个整数，输出其中的最小值。

Input

见Sample

Output

见Sample

Sample Input

3
12 6 1 90
10 40 30 20
-1 -3 -4 -5

Sample Output

min is 1
min is 10
min is -5

#代码
#include<stdio.h>
main()
{
int repeat,i,b,minz,min[4];
int a[4];
scanf(“%d”,&repeat);
if(repeat<10&&repeat>0)
{
for(b=0;b<repeat;b++)
{
for(i=0;i<4;i++)
{
scanf(“%d”,&a[i]);
}
minz=a[0];
for(i=0;i<4;i++)
{
if(a[i]<=minz)
minz=a[i];
}
min[b+1]=minz;
}
for(i=1;i<=3;i++)
{
printf(“min is %d\n”,min[i]);
}
}
}
