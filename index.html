1081:求奇数和

#题目
Description

输入一个正整数 repeat (0<repeat<10)，做repeat 次下列运算：
读入一批正整数(以零或负数为结束标志)，求其中的奇数和。

Sample Input

2
1 3 90 7 0
8 7 4 3 70 5 6 101 -1

Sample Output

11
116

#include <stdio.h>
int main( )
{
int a=1, sum=0,repeat,l;
scanf("%d", &repeat) ;
if(repeat<10&&repeat>0)
    {
    for(l=1;l<=repeat;l++)
        {
        while (a>0)
        {
        scanf("%d", &a) ;
        if(a<=0)
            {
            a=1;
            break;
            }	
        else if(a%2!=0)
        sum=sum+a ;
        }
        printf("%d\n",sum);
        sum=0;
        
        }
    }
    return 0 ;	
}

#题目
Description

读入 1 个正整数 n(n<=100)，计算并输出1－1/2＋1/3－1/4＋⋯⋯的前n 项和(保留3 位小数)。

Sample Input

1
3
Sample Output
0.833

#include<stdio.h>
main()
{
double sum=0;
int i,n,a,l,x=1;
scanf("%d",&a);
for(i=1;i<=a;i++)
{
    scanf("%d",&n);

}

1073:字母转换

#题目
Description

输入一个正整数 repeat (0<repeat<10)，做repeat 次下列运算：
输入一个字符，如果它是大写字母，输出相应的小写字母；如果它是小写字母，输出相应的
大写字母；否则，原样输出。

Sample Input

3F=y

Sample Output

f=Y 

#include<stdio.h>
main()
{
char a[3];
int repeat,i,l;
scanf("%d",&repeat);
    for(l=0;l<repeat;l++)
    {	
        scanf("%s",&a);
        for(i=0;i<=3;i++)
        {
        if(a[i]>='A'&&a[i]<='Z')
            a[i]=a[i]+32;
        else if(a[i]>='a'&&a[i]<='z')
            a[i]=a[i]-32;
        else 
            a[i]=a[i];
        }
        printf("%s",a);
    }
}

1072:显示水果的价格

#题目
Description

输入一个正整数 repeat (0<repeat<10)，做repeat 次下列运算：
以下 4 种水果的单价分别是3.00 元/公斤，2.50 元/公斤，4.10 元/公斤，10.20 元/公斤。
[1] apples
[2] pears
[3] oranges
[4] grapes
输入水果的编号，输出该水果的单价(保留2 位小数)。如果输入不正确的编号，显示单价为0.00。

Sample Input

1
3

Sample Output

price=4.10

#include<stdio.h>
main()
{
int a,b,i;
double price;
scanf("%d",&b);
for(i=0;i<b;i++)
    {
    scanf("%d",&a);
    switch(a)
        {
        case 1:printf("price=3.00\n");break;
        case 2:printf("price=2.50\n");break;
        case 3:printf("price=4.10\n");break;
        case 4:printf("price=10.20\n");break;
        default:printf("price=0.00\n");break;
        }
    }
}

1071:计算个人所得税

#题目
Description

输入一个正整数 repeat (0<repeat<10)，做repeat 次下列运算：
输入一个职工的月薪 salary，输出应交的个人所得税tax(保留2 位小数)。
tax = rate * (salary-850)
当 salary <= 850 时，rate = 0;
当 850 < salary <= 1350 时，rate = 5;
当 1350 < salary <= 2850 时，rate = 10;
当 2850 < salary <= 5850 时，rate = 15;
当 5850 < salary 时，rate = 20;

Sample Input

2
200
2000
Sample Output
tax=0.00
tax=115.00

HINT

当结果为0时，你的程序可能输出的是 -0.00，请判断为0时，使用printf(“tax=0.00\n”);

#include<stdio.h>
main()
{
double tax;
int b,rate,salary,i;
scanf("%d",&b);
    for(i=0;i<b;i++)
    {	
        scanf("%d",&salary);
        if(salary>5850)
        {
            rate=20;
        }	
        else if(salary>2850)
        {
            rate = 15;
        }
        else if(salary>1350)
        {
            rate=10;
        }	
        else if(salary>850)
        {
            rate=5;	
        }
        else
        {
            rate=0;
        }
        tax=rate*(salary-850)*0.01;
        if(tax==0)
        {
            printf("tax=0.00\n");
        }		
        else
        {
            printf("tax=%.2lf\n",tax);
        }
    }	
        return 0;
}
